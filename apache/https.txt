ETAPE 1:
openssl genrsa -des3 -out /etc/ssl/private/www.monsitetest.local.key 2048

ETAPE 2:
openssl req -new -key /etc/ssl/private/www.monsitetest.local.key -out /etc/ssl/private/www.monsitetest.local.csr

ETAPE 3:
openssl x509 -req -days 90 -in /etc/ssl/private/www.monsitetest.local.csr -signkey /etc/ssl/private/www.monsitetest.local.key -out /etc/ssl/private/www.monsitetest.local.cert

ETAPE 4:
openssl rsa -in /etc/ssl/private/www.monsitetest.local.key -out /etc/ssl/private/www.monsitetest.local.key.pem

a2enmod ssl
